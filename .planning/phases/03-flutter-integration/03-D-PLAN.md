```java
package com.termux.bridge;

import androidx.annotation.NonNull;

import com.termux.app.TermuxService;
import com.termux.app.TermuxSession;
import com.termux.terminal.TerminalSession;
import com.termux.shared.logger.Logger;

import java.util.ArrayList;
import java.util.List;

/**
 * Host-side implementation of TerminalBridgeApi (generated by Pigeon).
 * Bridges Flutter commands to the Termux terminal service.
 */
public class TerminalBridgeImpl implements TerminalBridgeApi {

    private static final String LOG_TAG = "TerminalBridgeImpl";
    private final TermuxService mTermuxService;

    public TerminalBridgeImpl(@NonNull TermuxService termuxService) {
        mTermuxService = termuxService;
    }

    @Override
    public void executeCommand(@NonNull String command) {
        TerminalSession currentSession = getCurrentTerminalSession();
        if (currentSession != null) {
            currentSession.write(command + "\n");
            Logger.logDebug(LOG_TAG, "Executed command: " + command);
        } else {
            Logger.logWarn(LOG_TAG, "No active terminal session for command execution");
        }
    }

    @NonNull
    @Override
    public String getTerminalOutput(@NonNull Long lines) {
        TerminalSession currentSession = getCurrentTerminalSession();
        if (currentSession == null || currentSession.getEmulator() == null) {
            return "";
        }

        String fullTranscript = currentSession.getEmulator().getScreen().getTranscriptText();
        if (fullTranscript.isEmpty()) return "";

        // Return last N lines
        String[] allLines = fullTranscript.split("\n");
        int startIndex = Math.max(0, allLines.length - lines.intValue());
        StringBuilder result = new StringBuilder();
        for (int i = startIndex; i < allLines.length; i++) {
            if (i > startIndex) result.append("\n");
            result.append(allLines[i]);
        }
        return result.toString();
    }

    @NonNull
    @Override
    public Long createSession() {
        TermuxSession newSession = mTermuxService.createTermuxSession(
            null, null, null, null, false, null
        );
        if (newSession != null) {
            int index = mTermuxService.getIndexOfSession(newSession.getTerminalSession());
            Logger.logDebug(LOG_TAG, "Created new session at index: " + index);
            return (long) index;
        }
        return -1L;
    }

    @NonNull
    @Override
    public List<SessionInfo> listSessions() {
        List<SessionInfo> result = new ArrayList<>();
        List<TermuxSession> sessions = mTermuxService.getTermuxSessions();
        for (int i = 0; i < sessions.size(); i++) {
            TermuxSession termuxSession = sessions.get(i);
            TerminalSession terminalSession = termuxSession.getTerminalSession();
            SessionInfo info = new SessionInfo.Builder()
                .setId((long) i)
                .setName(terminalSession.mSessionName != null
                    ? terminalSession.mSessionName
                    : "Session " + (i + 1))
                .setIsRunning(terminalSession.isRunning())
                .build();
            result.add(info);
        }
        return result;
    }

    private TerminalSession getCurrentTerminalSession() {
        if (mTermuxService.isTermuxSessionsEmpty()) return null;
        TermuxSession lastSession = mTermuxService.getLastTermuxSession();
        return lastSession != null ? lastSession.getTerminalSession() : null;
    }
}
```

Note: The exact method signatures (`Long` vs `long`, `@NonNull`, `Builder` pattern) depend on the Pigeon-generated interface. Read the generated `TerminalBridgeApi.java` file first and adapt the implementation to match exactly. The above is the logic — the types may need adjustment to match what Pigeon generates (e.g., Pigeon may use `Long` instead of `int` for numeric types, and may generate a `Builder` or constructor for `SessionInfo`).

Also verify `TerminalSession.mSessionName` field access — it may be accessed via a getter method. Check `TerminalSession.java` for the exact field/method name.

```java
import android.content.Context;
import android.content.pm.PackageManager;
import android.os.Build;

import androidx.annotation.NonNull;

import com.termux.shared.logger.Logger;

/**
 * Host-side implementation of SystemBridgeApi (generated by Pigeon).
 * Provides device and app metadata to Flutter.
 */
public class SystemBridgeImpl implements SystemBridgeApi {

    private static final String LOG_TAG = "SystemBridgeImpl";
    private final Context mContext;

    public SystemBridgeImpl(@NonNull Context context) {
        mContext = context.getApplicationContext();
    }

    @NonNull
    @Override
    public DeviceInfo getDeviceInfo() {
        return new DeviceInfo.Builder()
            .setManufacturer(Build.MANUFACTURER)
            .setModel(Build.MODEL)
            .setAndroidVersion(Build.VERSION.RELEASE)
            .setSdkInt((long) Build.VERSION.SDK_INT)
            .build();
    }

    @NonNull
    @Override
    public PackageInfo getPackageInfo() {
        String packageName = mContext.getPackageName();
        String versionName = "unknown";
        long versionCode = 0;

        try {
            android.content.pm.PackageInfo info =
                mContext.getPackageManager().getPackageInfo(packageName, 0);
            versionName = info.versionName;
            versionCode = info.getLongVersionCode();
        } catch (PackageManager.NameNotFoundException e) {
            Logger.logError(LOG_TAG, "Failed to get package info: " + e.getMessage());
        }

        return new PackageInfo.Builder()
            .setPackageName(packageName)
            .setVersionName(versionName)
            .setVersionCode(versionCode)
            .build();
    }
}
```

Note: As with TerminalBridgeImpl, the exact types and Builder pattern depend on Pigeon-generated code. Read `SystemBridgeApi.java` first and adapt. Pigeon may generate `DeviceInfo.Builder` or a constructor — match what's generated.

```java
package com.termux.bridge;

import android.os.Handler;
import android.os.Looper;

import androidx.annotation.NonNull;
import androidx.annotation.Nullable;

import com.termux.shared.logger.Logger;
import com.termux.terminal.TerminalSession;

import io.flutter.plugin.common.BinaryMessenger;

/**
 * Controls the SoulBridgeApi (Flutter API called from host side).
 * Handles debounced terminal output streaming to Flutter at max 10 updates/sec.
 */
public class SoulBridgeController {

    private static final String LOG_TAG = "SoulBridgeController";
    private static final long DEBOUNCE_DELAY_MS = 100; // 100ms = max 10 updates/sec

    private SoulBridgeApi mSoulBridge;
    private final Handler mDebounceHandler = new Handler(Looper.getMainLooper());
    private Runnable mPendingUpdate;

    /**
     * Initialize the SoulBridge with the Flutter engine's binary messenger.
     */
    public void setup(@NonNull BinaryMessenger binaryMessenger) {
        mSoulBridge = new SoulBridgeApi(binaryMessenger);
        Logger.logDebug(LOG_TAG, "SoulBridge connected to Flutter");
    }

    /**
     * Called when terminal text changes. Debounces output to max 10/sec.
     * Call this from TermuxTerminalSessionClient.onTextChanged().
     */
    public void onTerminalTextChanged(@Nullable TerminalSession session) {
        if (mSoulBridge == null || session == null || session.getEmulator() == null) return;

        // Cancel any pending debounced update
        if (mPendingUpdate != null) {
            mDebounceHandler.removeCallbacks(mPendingUpdate);
        }

        mPendingUpdate = () -> {
            try {
                String output = session.getEmulator().getScreen().getTranscriptText();
                // Send last portion to avoid sending massive transcript
                String[] lines = output.split("\n");
                int startIndex = Math.max(0, lines.length - 50);
                StringBuilder lastLines = new StringBuilder();
                for (int i = startIndex; i < lines.length; i++) {
                    if (i > startIndex) lastLines.append("\n");
                    lastLines.append(lines[i]);
                }
                mSoulBridge.onTerminalOutput(lastLines.toString(), reply -> {});
            } catch (Exception e) {
                Logger.logError(LOG_TAG, "Failed to stream terminal output: " + e.getMessage());
            }
        };

        mDebounceHandler.postDelayed(mPendingUpdate, DEBOUNCE_DELAY_MS);
    }

    /**
     * Notify Flutter of a session change.
     */
    public void onSessionChanged(int sessionId, @NonNull String sessionName, boolean isRunning) {
        if (mSoulBridge == null) return;

        SessionInfo info = new SessionInfo.Builder()
            .setId((long) sessionId)
            .setName(sessionName)
            .setIsRunning(isRunning)
            .build();

        mSoulBridge.onSessionChanged(info, reply -> {});
    }

    /**
     * Clean up resources.
     */
    public void teardown() {
        if (mPendingUpdate != null) {
            mDebounceHandler.removeCallbacks(mPendingUpdate);
        }
        mSoulBridge = null;
    }
}
```

Read the generated `TerminalBridgeApi.java` (which contains `SoulBridgeApi` since they're in the same Pigeon file) to confirm the exact constructor signature and callback type. It may be `SoulBridgeApi(BinaryMessenger)` or have a different constructor. The `reply` callback type may be `SoulBridgeApi.Reply<Void>` or similar.

```java
private void setupPigeonBridges() {
    FlutterEngine flutterEngine = FlutterEngineCache.getInstance()
        .get(TermuxApplication.FLUTTER_ENGINE_ID);

    if (flutterEngine == null || mTermuxService == null) {
        Logger.logWarn(LOG_TAG, "Cannot setup Pigeon bridges: engine or service is null");
        return;
    }

    io.flutter.plugin.common.BinaryMessenger messenger =
        flutterEngine.getDartExecutor().getBinaryMessenger();

    // Register host APIs (Flutter → Host)
    TerminalBridgeApi.setUp(messenger, new TerminalBridgeImpl(mTermuxService));
    SystemBridgeApi.setUp(messenger, new SystemBridgeImpl(this));

    // Setup Flutter API (Host → Flutter)
    mSoulBridgeController = new SoulBridgeController();
    mSoulBridgeController.setup(messenger);

    Logger.logDebug(LOG_TAG, "Pigeon bridges registered");
}
```
