Create `flutter_module/lib/ui/chat/widgets/terminal_command_indicator.dart`:

```dart
import 'package:flutter/material.dart';
import '../../../generated/terminal_bridge.g.dart';

class TerminalCommandIndicator extends StatelessWidget {
  final String commandName;
  final VoidCallback? onTap;

  const TerminalCommandIndicator({
    super.key,
    required this.commandName,
    this.onTap,
  });

  @override
  Widget build(BuildContext context) {
    final colorScheme = Theme.of(context).colorScheme;
    return GestureDetector(
      onTap: onTap ?? () => _openTerminalSheet(),
      child: Container(
        margin: const EdgeInsets.symmetric(horizontal: 16, vertical: 8),
        padding: const EdgeInsets.symmetric(horizontal: 16, vertical: 12),
        decoration: BoxDecoration(
          color: colorScheme.surfaceContainerHigh,
          borderRadius: BorderRadius.circular(12),
          border: Border.all(
            color: const Color(0xFF6C63FF).withOpacity(0.3),
          ),
        ),
        child: Row(
          mainAxisSize: MainAxisSize.min,
          children: [
            SizedBox(
              width: 16,
              height: 16,
              child: CircularProgressIndicator(
                strokeWidth: 2,
                color: const Color(0xFF6C63FF),
              ),
            ),
            const SizedBox(width: 12),
            Flexible(
              child: Column(
                crossAxisAlignment: CrossAxisAlignment.start,
                mainAxisSize: MainAxisSize.min,
                children: [
                  Text(
                    'Uitvoeren in terminal...',
                    style: TextStyle(
                      color: colorScheme.onSurface,
                      fontSize: 13,
                      fontWeight: FontWeight.w500,
                    ),
                  ),
                  const SizedBox(height: 2),
                  Text(
                    commandName,
                    style: TextStyle(
                      color: colorScheme.onSurfaceVariant,
                      fontSize: 12,
                      fontFamily: 'monospace',
                    ),
                    maxLines: 1,
                    overflow: TextOverflow.ellipsis,
                  ),
                ],
              ),
            ),
            const SizedBox(width: 8),
            Icon(
              Icons.terminal,
              size: 18,
              color: const Color(0xFF6C63FF),
            ),
          ],
        ),
      ),
    );
  }

  Future<void> _openTerminalSheet() async {
    final bridge = TerminalBridgeApi();
    await bridge.openTerminalSheet();
  }
}
```

Key properties:
- SOUL purple accent `#6C63FF` for border, spinner, and icon
- `commandName` shown in monospace font (truncated with ellipsis)
- `"Uitvoeren in terminal..."` as header text (Dutch per user preferences)
- `Icons.terminal` icon on the right
- Tap opens terminal sheet via `TerminalBridgeApi().openTerminalSheet()` (AWAR-08)
- `GestureDetector` wraps entire card for tap target
